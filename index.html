Confirmation

            
bimPN.confirm({
  title: 'Delete Data',
  text: 'Do you want to delete?',
  icon: 'warning',
  confirmButtonText: 'Yes',
  cancelButtonText: 'No',
})
.then((res) => {
  console.log(res);
  if(res.isConfirmed){
    bimPN.alert({
      title: 'Success!',
      text: 'Data deleted',
      icon: 'check',
      location: 'center',
    });
  }
  else{
    bimPN.alert({
      title: 'Cancelled',
      text: 'Data retained',
      location: 'center',
    });
  }
});
